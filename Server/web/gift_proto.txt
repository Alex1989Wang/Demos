#所有的http协议必须有头部必须包含

request header字段中
X-Device-Token: 设备token 在用户不授权的时候，ios可能拿不到device_token，所以ios中此处用uuid
X-App-Token: 逻辑token
X-App-Uid: 逻辑uid
X-Device: 设备名称 Xiaomi Redmi Note 2
X-OS: 操作系统信息 Android 5.0.2
X-App-Version: app版本 2.5.0
X-Locale: 所在地 zh_CN
X-Appid: appid 10001
X-Anonymous-User: 是否匿名 1:是 0:否


response http状态码
200 操作成功
业务状态码见 svn://120.24.63.167/server/src/common/http_errcodes.go





currency_info                //json货币信息
{
    amount:                  //int64数量  单位:厘
    currency:                //int 币种  1：金币 2:代币(人民币) 3:xx
}

gift_info                    //json礼物信息 
{
    gift_id:                 //int  
    gift_name:                  //stirng 礼物名称
    gift_price:               //currency_info 礼物价格
    gift_icon:               //string 礼物图片 
}

gift_broadcast_baseinfo   //json礼物广播信息
{
    send_uinfo:            //room_userinfo 
    to_uinfo:              //room_userinfo
    room_id:               //uint64 房间id
    gift_id:               //uint64 礼物id
    all_group_gift_count   //uint64 已经送礼物个数
    server_group_id:       //string 服务端连送id
    client_group_id:       //string 客户端连送id
}  

gift_broadcast_info          //json礼物广播信息
{
    category:              //string 类别 这里是room
    type:                  //int    类别下分类 这里是22
    data:[gift_broadcast_baseinfo...] //广播基本信息 可能有多条

}


--
获取礼物信息
--
/gifts

方法:           GET
                    
返回:           json 格式
                {
                    code:                              // int 错误代码
                        HTTP_EC_SUCCESS
                        HTTP_EC_SYS_ERROR 
                        HTTP_EC_UNKNOWN_ERROR
                        HTTP_EC_INVALID_ARGS                
                    msg:                               // string 错误消息
                    data:[gift_info ...]               // 列表 所有的礼物信息
                }
                                
--
送礼物
--
/room/gifts/:to_uid      :to_uid//接受者uid
方法:           POST

参数:          gift_id:                     //int 
               gift_count                  //int 礼物数量
               room_id                   //uint64 房间id
               send_id                   //string 客户端生成对于uid唯一id,防重复发送,便于客户的和服务端对账
               client_group_id           //string 客户的group_id
               
返回:           json 格式
                {
                    code:                                     // int 错误代码
                        HTTP_EC_SUCCESS
                        HTTP_EC_SYS_ERROR 
                        HTTP_EC_UNKNOWN_ERROR
                        HTTP_EC_INVALID_ARGS
                        HTTP_EC_INSUFFICIENT_BALANCE
                        HTTP_EC_IVALID_TO_UID                                    
                    msg:                                       // string 错误消息

                }
